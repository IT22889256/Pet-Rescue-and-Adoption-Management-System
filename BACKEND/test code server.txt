const express = require("express");
require('dotenv').config(); // Importing dotenv for environment variables
const mongoose = require("mongoose");
const employeeRoute = require("./routes/employee.route.js"); // Importing employee route
const attendanceRoute = require("./routes/attendance.route.js"); // Importing attendance route
const salaryRoute = require("./routes/salary.route.js"); // Importing salary route
const jobRoleRoute = require("./routes/jobRole.route.js"); // Importing job role route
const app = express();

// Middleware
app.use(express.json()); // Parsing incoming requests with JSON payloads
app.use(express.urlencoded({ extended: false })); // Parsing incoming requests with URL-encoded payloads

// Routes
app.use("/api/employees", employeeRoute); // Employee routes
app.use("/api/attendance", attendanceRoute); // Attendance routes
app.use("/api/salary", salaryRoute); // Salary routes
app.use("/api/jobRole", jobRoleRoute); // Job role routes

// Home route
app.get("/", (req, res) => {
  res.send("Hello from Node API Server Updated");
});

// Connect to database
mongoose
  .connect(process.env.MONGODB_URL_TEST) // Connecting to MongoDB using environment variable
  .then(() => {
    console.log("Connected to database!");
    app.listen(3000, () => {
      console.log("Server is running on port 3000");
    });
  })
  .catch(() => {
    console.log("Connection failed!");
  });
